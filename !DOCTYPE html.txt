    <script>
        // Asset definitions with correct quadrants
        const assets = [
            { id: 'savings', name: 'Savings Account', icon: 'fas fa-piggy-bank', color: '#3b82f6', quadrant: 'low-risk-low-return' },
            { id: 'gov-bonds', name: 'Government Bonds', icon: 'fas fa-university', color: '#1e40af', quadrant: 'low-risk-low-return' },
            { id: 'corp-bonds', name: 'Corporate Bonds', icon: 'fas fa-building', color: '#6366f1', quadrant: 'low-risk-low-return' },
            { id: 'blue-chip', name: 'Blue Chip Stocks', icon: 'fas fa-star', color: '#8b5cf6', quadrant: 'low-risk-high-return' },
            { id: 'real-estate', name: 'Real Estate', icon: 'fas fa-home', color: '#10b981', quadrant: 'low-risk-high-return' },
            { id: 'gold', name: 'Gold', icon: 'fas fa-coins', color: '#f59e0b', quadrant: 'low-risk-high-return' },
            { id: 'growth-stocks', name: 'Growth Stocks', icon: 'fas fa-rocket', color: '#ef4444', quadrant: 'high-risk-high-return' },
            { id: 'commodities', name: 'Commodities', icon: 'fas fa-oil-can', color: '#dc2626', quadrant: 'high-risk-high-return' },
            { id: 'cryptocurrency', name: 'Cryptocurrency', icon: 'fab fa-bitcoin', color: '#FF8C00', quadrant: 'high-risk-high-return' },
            { id: 'emerging-markets', name: 'Emerging Markets', icon: 'fas fa-globe-americas', color: '#7c3aed', quadrant: 'high-risk-high-return' },
            { id: 'options', name: 'Options', icon: 'fas fa-exchange-alt', color: '#be185d', quadrant: 'high-risk-low-return' },
            { id: 'futures', name: 'Futures', icon: 'fas fa-chart-bar', color: '#991b1b', quadrant: 'high-risk-low-return' }
        ];

        let score = 0;
        let correctPlacements = 0;
        let draggedAsset = null;

        // Initialize the game
        function initGame() {
            createAssets();
            setupDragAndDrop();
            updateScore();
        }

        // Create asset elements
        function createAssets() {
            const container = document.getElementById('assets-container');
            container.innerHTML = '';
            
            // Shuffle assets for random placement
            const shuffledAssets = [...assets].sort(() => Math.random() - 0.5);
            
            shuffledAssets.forEach(asset => {
                const assetElement = document.createElement('div');
                assetElement.className = 'asset-item bg-white rounded-full w-20 h-20 flex flex-col items-center justify-center shadow-md border-2';
                assetElement.draggable = true;
                assetElement.dataset.assetId = asset.id;
                assetElement.dataset.quadrant = asset.quadrant;
                assetElement.style.borderColor = asset.color;
                
                assetElement.innerHTML = `
                    <i class="${asset.icon} text-lg mb-1" style="color: ${asset.color}"></i>
                    <span class="text-xs text-center font-medium text-gray-700 leading-tight">${asset.name.split(' ').join('<br>')}</span>
                `;
                
                container.appendChild(assetElement);
            });
        }

        // Setup drag and drop functionality
        function setupDragAndDrop() {
            const assetElements = document.querySelectorAll('.asset-item');
            const dropZones = document.querySelectorAll('.drop-zone');

            assetElements.forEach(asset => {
                asset.addEventListener('dragstart', handleDragStart);
                asset.addEventListener('dragend', handleDragEnd);
            });

            dropZones.forEach(zone => {
                zone.addEventListener('dragover', handleDragOver);
                zone.addEventListener('dragenter', handleDragEnter);
                zone.addEventListener('dragleave', handleDragLeave);
                zone.addEventListener('drop', handleDrop);
            });
        }

        function handleDragStart(e) {
            draggedAsset = e.target;
            e.target.classList.add('dragging');
        }

        function handleDragEnd(e) {
            e.target.classList.remove('dragging');
            draggedAsset = null;
        }

        function handleDragOver(e) {
            e.preventDefault();
        }

        function handleDragEnter(e) {
            e.preventDefault();
            e.target.classList.add('drag-over');
        }

        function handleDragLeave(e) {
            e.target.classList.remove('drag-over');
        }

        function handleDrop(e) {
            e.preventDefault();
            e.target.classList.remove('drag-over');
            
            if (!draggedAsset) return;
            
            const assetQuadrant = draggedAsset.dataset.quadrant;
            const targetQuadrant = e.target.dataset.quadrant;
            
            // Remove asset from current position
            draggedAsset.remove();
            
            // Create new asset in drop zone
            const newAsset = draggedAsset.cloneNode(true);
            newAsset.style.position = 'absolute';
            newAsset.style.left = `${Math.random() * (e.target.offsetWidth - 80)}px`;
            newAsset.style.top = `${Math.random() * (e.target.offsetHeight - 80)}px`;
            
            // Check if placement is correct
            if (assetQuadrant === targetQuadrant) {
                e.target.classList.add('correct-placement');
                setTimeout(() => e.target.classList.remove('correct-placement'), 1000);
                score += 10;
                correctPlacements++;
            } else {
                e.target.classList.add('incorrect-placement');
                setTimeout(() => e.target.classList.remove('incorrect-placement'), 1000);
                score = Math.max(0, score - 2);
            }
            
            e.target.appendChild(newAsset);
            
            // Re-setup drag and drop for new asset
            newAsset.addEventListener('dragstart', handleDragStart);
            newAsset.addEventListener('dragend', handleDragEnd);
            
            updateScore();
            checkGameComplete();
        }

        function updateScore() {
            document.getElementById('score').textContent = score;
            document.getElementById('correct-count').textContent = correctPlacements;
        }

        function checkGameComplete() {
            if (correctPlacements === assets.length) {
                setTimeout(() => {
                    alert(`ðŸŽ‰ Congratulations! You've completed the game with a score of ${score}!`);
                }, 500);
            }
        }

        function resetGame() {
            score = 0;
            correctPlacements = 0;
            
            // Clear all drop zones
            document.querySelectorAll('.drop-zone').forEach(zone => {
                const assets = zone.querySelectorAll('.asset-item');
                assets.forEach(asset => asset.remove());
            });
            
            // Recreate assets in the pool
            createAssets();
            setupDragAndDrop();
            updateScore();
        }

        // Event listeners
        document.getElementById('reset-btn').addEventListener('click', resetGame);

        // Initialize the game when page loads
        document.addEventListener('DOMContentLoaded', initGame);
    </script>
